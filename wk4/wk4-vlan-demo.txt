

!!S1 config

en
config t
host S1

no ip domain-loo

line con 0
 logg syn

username admin1 pri 15 sec letmein

line vty 0 4
  login local
  transport input telnet
  exec-timeout 0 5

vlan 10
name HR
vlan 20
name SALES

int range f0/1-10, g0/2
  switchport mode access
  switchport access vlan 10

int range f0/11-20, f0/24
  switchport mode access
  switchport access vlan 20

int g0/1
  switchport mode trunk
  switchport trunk native vlan 10
  switchport trunk allowed vlan 1,10,20
